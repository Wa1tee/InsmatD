\begin{center}
\begin{align*}
	y' = (x + 1)(y + 1) &<=> \dfrac{dy}{dx} = (x + 1)(y + 1) \\
						&<=> \dfrac{1}{y+1} dy = (x+1) dx \\
						&<=> \int \dfrac{1}{y+1}dy = \int (x+1) dx + C \\
						&<=> ln(y+1) = \dfrac{1}{2}x^2 + x + C \\
						&<=> y+1 = e^{\frac{1}{2} x^2 + x + C} \\
						&<=> y = e^{\frac{1}{2} x^2 + x + C}
\end{align*}
\end{center}
